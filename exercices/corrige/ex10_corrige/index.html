    DROP PROCEDURE IF EXISTS concat_pays;
DELIMITER $$
CREATE PROCEDURE concat_pays (OUT liste_pays VARCHAR(1000))
BEGIN

    DECLARE fin_enregistrement INTEGER DEFAULT 0;
    DECLARE nom VARCHAR(30);
    DECLARE curseur_pays CURSOR FOR
        SELECT pays FROM houblon GROUP BY pays ORDER BY pays;

    DECLARE CONTINUE HANDLER FOR NOT FOUND SET fin_enregistrement = 1;

    SET liste_pays = "";

    OPEN curseur_pays;

    loop_pays: LOOP
        FETCH curseur_pays INTO nom;
        IF fin_enregistrement = 1 THEN
        LEAVE loop_pays;
        END IF;

        SET liste_pays = CONCAT(liste_pays, ";", nom);
    END LOOP loop_pays;

    -- Supprime le premier caractère, un point-virgule
    SET liste_pays = SUBSTR(liste_pays, 2);

    CLOSE curseur_pays;
END$$
DELIMITER ;

CALL concat_pays(@listePays);
SELECT @listePays;

    DROP TABLE IF EXISTS similitude;
CREATE TABLE similitude (
    houblon_id INTEGER,
    similitude_id INTEGER,
    PRIMARY KEY(houblon_id, similitude_id),
    CONSTRAINT `houblon_fk` FOREIGN KEY (houblon_id) REFERENCES houblon (id) ON UPDATE RESTRICT ON DELETE CASCADE,
    CONSTRAINT `similitude_fk` FOREIGN KEY (similitude_id) REFERENCES houblon (id) ON UPDATE RESTRICT ON DELETE CASCADE
);

DROP TABLE IF EXISTS similitude_log;
CREATE TABLE similitude_log (
    id INTEGER AUTO_INCREMENT,  
    houblon_id INTEGER,
    nom VARCHAR(50),
    PRIMARY KEY(id),
    CONSTRAINT `houblon_log_fk` FOREIGN KEY (houblon_id) REFERENCES houblon (id) ON UPDATE RESTRICT ON DELETE CASCADE
);


DROP PROCEDURE IF EXISTS extract_similitude;
DELIMITER $$
CREATE PROCEDURE extract_similitude()
BEGIN

    DECLARE fin_enregistrement INTEGER DEFAULT 0;
    -- Variable pour le curseur
    DECLARE id_houblon INTEGER;
    DECLARE similitude VARCHAR(200);
    -- Variable de la deuxième boucle
    DECLARE nom_suggestion VARCHAR(50);
    DECLARE id_houblon_suggestion INT;
    DECLARE position_separateur INT;

    DECLARE curseur_houblon CURSOR FOR
        SELECT id, similitude FROM houblon h WHERE NOT similitude IS NULL;

    -- Condition de sortie du curseur
    DECLARE CONTINUE HANDLER FOR NOT FOUND SET fin_enregistrement = 1;

    -- Pour attraper les erreurs SQLEXCEPTION
    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        ROLLBACK;
        SELECT 'Une erreur est survenue, les opérations ont été annulé.';
    END;

    START TRANSACTION;

    OPEN curseur_houblon;

    -- Boucle qui parcourt le curseur
    loop_houblon: LOOP
        -- Récupère les valeurs de la ligne active du curseur
        FETCH curseur_houblon INTO id_houblon, similitude;

        IF fin_enregistrement = 1 THEN
            LEAVE loop_houblon;
        END IF;

        -- Extrait chaque valeur séparé par une virgule du champs similitude
        loop_suggestions: LOOP

            -- Détecte la position du séparateur 
            SET position_separateur = INSTR(similitude, ',');
            -- Isole le nom du houblon qui est devant le premier séparateur trouvé
            SET nom_suggestion = SUBSTRING_INDEX(similitude, ',', 1);
            -- Sélectionne le id du houblon ou retourne NULL s'il n'est pas trouvé
            SELECT IFNULL((SELECT id FROM houblon WHERE nom LIKE CONCAT('%',nom_suggestion,'%')), NULL) INTO id_houblon_suggestion;     

            IF NOT id_houblon_suggestion IS NULL THEN
                -- Si trouvé, l'ajoute dans la table similitude
                INSERT INTO similitude(houblon_id, similitude_id) VALUES(id_houblon, id_houblon_suggestion);
            ELSE
                -- Si non, l'ajoute dans la table similitude_log
                INSERT INTO similitude_log(houblon_id, nom) VALUES(id_houblon, nom_suggestion);
            END IF;

            -- Conserve le texte après le séparateur pour la prochaine itération
            SELECT trim(substring(similitude, position_separateur + 1)) INTO similitude; 


            -- Condition de sortie
            IF (position_separateur) = 0 THEN
                LEAVE loop_suggestions;
            END IF;
            LEAVE loop_suggestions;
        END LOOP loop_suggestions;


    END LOOP loop_houblon;

    CLOSE curseur_houblon;

    COMMIT;
END$$
DELIMITER ;

CALL extract_similitude();
